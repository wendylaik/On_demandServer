package archivo;

import java.io.BufferedWriter;
import java.io.FileWriter;
import java.io.IOException;


public class Cabeza {
    
    Nodoo cabeza;
    
    public Cabeza() {
        this.cabeza = null;
    }
    
    public void mostrar() {
        Nodoo temp = cabeza;
        while (temp != null) {
            System.out.println(temp.palabra);
            temp = temp.Primero;
        }
    }
    
    
      public void insertar(String palabra) {
        Nodoo nuevoNodo = new Nodoo(palabra);
        if (cabeza == null) {
            cabeza = nuevoNodo;
        } else {
            Nodoo temp = cabeza;
            while (temp.Primero != null) {
                temp = temp.Primero;
            }
            temp.Primero = nuevoNodo;
        }
    }
    
     public void eliminar(String palabra) {
        if (cabeza != null) {
            if (cabeza.palabra.equals(palabra)) {
                cabeza = cabeza.Primero;
            } else {
                Nodoo temp = cabeza;
                while (temp.Primero != null && !temp.Primero.palabra.equals(palabra)) {
                    temp = temp.Primero;
                }
                if (temp.Primero != null) {
                    temp.Primero = temp.Primero.Primero;
                }
            }
        } 
     }
     public void escribirEnArchivo(String nombreArchivo) {
        try (BufferedWriter writer = new BufferedWriter(new FileWriter(nombreArchivo))) {
            Nodoo temp = cabeza;
            while (temp != null) {
                writer.write(temp.palabra);
                writer.newLine();
                temp = temp.Primero;
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
     
     
     public class Nodoo {
    
    String palabra;
    Nodoo Primero;

    public Nodoo(String palabra) {
        this.palabra = palabra;
        this.Primero = null;
    }
    
    
}


Cabeza lista = new Cabeza();
        String nombreArchivo = "Palabras.txt";
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        String opcion = "";
        
        // Leer el archivo y agregar palabras a la lista enlazada
       try (BufferedReader br = new BufferedReader(new FileReader(nombreArchivo))) {
            String linea;
            while ((linea = br.readLine()) != null) {
                lista.insertar(linea);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }

        do {
            System.out.println("\nMENU:");
            System.out.println("1. Mostrar lista");
            System.out.println("2. Añadir palabra");
            System.out.println("3. Eliminar palabra");
            System.out.println("4. Guardar y salir");
            System.out.print("Seleccione una opción: ");

            try {
                opcion = reader.readLine();
                switch (opcion) {
                    case "1":
                        System.out.println("\nLista:");
                        lista.mostrar();
                        break;
                    case "2":
                        System.out.print("Ingrese la palabra a añadir: ");
                        String nuevaPalabra = reader.readLine();
                        lista.insertar(nuevaPalabra);
                        break;
                    case "3":
                        System.out.print("Ingrese la palabra a eliminar: ");
                        String palabraEliminar = reader.readLine();
                        lista.eliminar(palabraEliminar);
                        break;
                    case "4":
                        System.out.println("\nGuardando y saliendo...");
                        lista.escribirEnArchivo(nombreArchivo);
                        break;
                    default:
                        System.out.println("\nOpción inválida, por favor seleccione una opción válida.");
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        } while (!opcion.equals("4"));
    }

        
}
